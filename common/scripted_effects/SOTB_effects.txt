# --- Phase 0: Foundation Effects ---

# Remove Delchev's watching spirit (triggers when succession resolves)
SOTB_remove_delchev_watches = {
	remove_ideas = {
		MAC_founder_watches
	}
}

# Goce Delchev returns as country leader
SOTB_delchev_returns = {
	set_country_leader = {
		ideology = populism
		traits = { delchevism_founder }
		id = -1
	}
	remove_ideas = {
		MAC_founder_watches
		MAC_lion_watches
	}
	add_ideas = {
		MAC_delchev_returns
	}
}

# Jane Sandanski takes the Lion's Path
SOTB_sandanski_path = {
	set_country_leader = {
		ideology = conservatism
		traits = { lion_of_solun }
		id = -1
	}
	remove_ideas = {
		MAC_founder_watches
		MAC_lion_watches
	}

		MAC_lion_leads
	}
}

# Todor Aleksandrov seizes power (Non-Aligned military)

	}

	}
}

# Ivan Mihailov's fascist takeover
SOTB_mihailov_coup = {

	}

	}
}
