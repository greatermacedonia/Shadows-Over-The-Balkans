# The Founder Sends a Letter
country_event = {
	id = mac_delcev.1
	title = mac_delcev.1.t
	desc = mac_delcev.1.d
	picture = GFX_report_event_macedonian_politics
	
	is_triggered_only = yes
	
	option = {
		name = mac_delcev.1.a  # "What does he say?"
		
		custom_effect_tooltip = MAC_delcev_evaluates_tt
		
		# Delčev evaluates your political choice
		if = {
			limit = { has_completed_focus = MAC_vmro_conspiracy }
			add_stability = -0.02
			country_event = { id = mac_delcev.3 days = 7 }  # Delčev disapproves of fascism
		}
		else_if = {
			limit = { has_completed_focus = MAC_workers_strike }
			add_stability = -0.01
			country_event = { id = mac_delcev.4 days = 7 }  # Delčev wary of communism
		}
		else_if = {
			limit = { has_completed_focus = MAC_balkan_conference }
			add_stability = 0.02
			country_event = { id = mac_delcev.5 days = 7 }  # Delčev approves of democracy
		}
		else_if = {
			limit = { has_completed_focus = MAC_emergency_powers }
			add_stability = 0.01
			country_event = { id = mac_delcev.6 days = 7 }  # Delčev neutral on centrism
		}
	}
}

			OR = {

			}
		}
		
		add_ideas = MAC_delcev_dream_fulfilled
		add_stability = 0.15
		add_war_support = 0.10

		# Successor takes over
		if = {
			limit = { has_country_flag = MAC_successor_vukmanovic }
			promote_character = MAC_vukmanovic
		}
		else_if = {
			limit = { has_country_flag = MAC_successor_apostolski }
			promote_character = MAC_apostolski
		}
		else_if = {
			limit = { has_country_flag = MAC_successor_vlahov }
			promote_character = MAC_vlahov
		}
	}
	
	# He condemns your government
	option = {
		name = mac_delcev.71.b
		
		trigger = {
			OR = {
				has_country_flag = MAC_delcev_disapproves
				is_subject = yes
				is_in_faction_with = SOV
				is_in_faction_with = GER
			}
		}
		
		add_ideas = MAC_delcev_disappointment
		add_stability = -0.10
		add_war_support = -0.05

		NOT = { has_country_flag = delcev_dead }
		date > 1940.1.1
	}
	
	mean_time_to_happen = {
		months = 36
		modifier = {
			factor = 0.5
			date > 1943.1.1  # Higher chance if game goes long
		}
		modifier = {
			factor = 0.3
			date > 1945.1.1
		}
	}
	
	fire_only_once = yes
	
	option = {
		name = mac_delcev.91.a  # "All of Macedonia mourns..."
		add_stability = -0.20
		add_war_support = -0.15
		add_political_power = -100
		set_country_flag = delcev_dead
		
		if = {
			limit = {
				any_character = {
					is_country_leader = yes
					has_trait = living_legend
				}
			}
			remove_country_leader_trait = living_legend
		}
		
		custom_effect_tooltip = mac_delcev.91.a_tt
		
		# Succession crisis or smooth transition
		if = {
			limit = { has_country_flag = delcev_succession_settled }
			country_event = { id = mac_delcev.92 days = 7 }  # Smooth transition
		}
		else = {
			country_event = { id = mac_delcev.93 days = 7 }  # Succession crisis
		}
	}
}
