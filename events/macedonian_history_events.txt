	id = macedonia_history.3
	title = macedonia_history.3.t
	desc = macedonia_history.3.d
	picture = GFX_report_event_generic_factory
	
	trigger = {
		tag = MAC
		has_idea = MAC_ottoman_legacy
		OR = {
			has_political_power > 100
			date > 1937.1.1
		}

	}

	}
}

# Event: Yugoslav Federation Proposal
country_event = {
	id = macedonia_history.10
	title = macedonia_history.10.t
	desc = macedonia_history.10.d
	picture = GFX_report_event_generic_sign_treaty1
	
	trigger = {
		tag = MAC
		country_exists = YUG
		NOT = { has_war_with = YUG }
		has_government = communism
		YUG = { has_government = communism }
		NOT = { has_country_flag = yugoslav_federation_asked }
	}
	
	mean_time_to_happen = {
		days = 180
	}
	
	fire_only_once = yes
	
	option = {
		name = macedonia_history.10.a  # Join Yugoslavia
		set_country_flag = yugoslav_federation_asked
		YUG = { country_event = { id = macedonia_history.11 days = 1 } }
	}
	
	option = {
		name = macedonia_history.10.b  # Maintain independence
		set_country_flag = yugoslav_federation_asked
		add_political_power = 100
		add_stability = -0.05
	}
}
